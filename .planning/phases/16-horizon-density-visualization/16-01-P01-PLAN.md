```typescript
import { computeNextDue, type Task, type Completion } from '@/lib/task-scheduling';
import { formatInTimeZone } from 'date-fns-tz';

export type ShiftInfo = {
  ideal: Date | null;
  scheduled: Date | null;
  displaced: boolean;
};

/**
 * Phase 16 (LVIZ-03, LVIZ-04, D-04, D-10): Pure helper returning both the
 * natural ideal and the (possibly LOAD-smoothed) scheduled date for a task.
 *
 * Algorithm:
 *   - scheduled = computeNextDue(task, lastCompletion, now, undefined, timezone)
 *   - ideal     = computeNextDue({...task, next_due_smoothed: null}, lastCompletion, now, undefined, timezone)
 *   - displaced = ideal && scheduled && Math.abs(daysDiff) >= 1 (LVIZ-04: >0 days)
 *
 * LOAD-06 (D-04): anchored-mode tasks never smooth → ideal === scheduled for
 * them by computeNextDue's schedule_mode guard (see task-scheduling.ts:256).
 * We rely on that branch — no explicit anchored short-circuit here.
 *
 * Null cases (ideal=null or scheduled=null): displaced=false. Covers archived,
 * dormant-seasonal, completed-OOFT — all surfaces that should never badge.
 *
 * Displacement threshold: Math.abs(differenceInCalendarDays(scheduled, ideal)) >= 1.
 * A sub-day tz-edge drift does NOT qualify as displacement (prevents noise
 * from DST shifts). Use differenceInCalendarDays (not raw ms diff) for
 * calendar-day precision in home tz.
 */
export function getIdealAndScheduled(
  task: Task,
  lastCompletion: Completion | null,
  now: Date,
  timezone?: string,
): ShiftInfo {
  const scheduled = computeNextDue(task, lastCompletion, now, undefined, timezone);
  const naturalTask: Task = { ...task, next_due_smoothed: null };
  const ideal = computeNextDue(naturalTask, lastCompletion, now, undefined, timezone);

  if (!ideal || !scheduled) {
    return { ideal, scheduled, displaced: false };
  }

  // Calendar-day diff in home tz — use differenceInCalendarDays from date-fns
  // on the UTC instants (date-fns computes in UTC; ≥1 day qualifies as
  // displaced per LVIZ-04 "> 0 days"). Sub-day drift (DST, rounding) does
  // NOT count as displacement.
  const diffDays = Math.abs(
    Math.round((scheduled.getTime() - ideal.getTime()) / 86400000)
  );
  return { ideal, scheduled, displaced: diffDays >= 1 };
}

/**
 * Phase 16 (LVIZ-01, D-01): Pure month-density counter.
 *
 * Returns Map&lt;monthKey, count&gt; where monthKey = formatInTimeZone(nextDue, tz, 'yyyy-MM').
 * Only counts tasks whose scheduled next-due is non-null (so dormant seasonal +
 * archived + completed-OOFT + invalid-freq tasks are all excluded upstream by
 * computeNextDue returning null).
 *
 * The HorizonStrip consumes this Map alongside its own 12-month key loop —
 * months with no key in the Map render with no tint (bg-primary/0 default).
 *
 * O(N) in tasks. Stateless; safe to call per render. Perf budget validated
 * by Phase 12-04 (100-task placement ~4ms; density is strictly cheaper).
 *
 * NOTE: caller must provide latestByTask so we can reuse the already-fetched
 * completion Map (no duplicate PB roundtrip).
 */
export function computeMonthDensity(
  tasks: Task[],
  latestByTask: Map<string, { completed_at: string }>,
  now: Date,
  timezone: string,
): Map<string, number> {
  const out = new Map<string, number>();
  for (const task of tasks) {
    if (task.archived) continue;
    const last = latestByTask.get(task.id) ?? null;
    const next = computeNextDue(task, last, now, undefined, timezone);
    if (!next) continue;
    const key = formatInTimeZone(next, timezone, 'yyyy-MM');
    out.set(key, (out.get(key) ?? 0) + 1);
  }
  return out;
}
```

import { formatInTimeZone } from 'date-fns-tz';

/**
 * Phase 16 (LVIZ-03, LVIZ-04, D-05, D-06): shift badge for tasks whose
 * scheduled next-due differs from their natural ideal by ≥1 calendar day.
 *
 * Emoji ⚖️ (balance scale) is the load-smoothing semantic — "we rebalanced
 * this". Inline <span> render (no shadcn dep, no new lucide icon).
 *
 * Tooltip via `title` attribute — native browser tooltip. Phase 16
 * explicitly avoids adding a radix-tooltip dep (D-05 Claude's Discretion:
 * use Tooltip if already installed, else inline title; inline title ships
 * everywhere, zero weight added).
 *
 * Placements (D-06): TaskRow, TaskBand variants, TaskDetailSheet header,
 * HorizonStrip Sheet drawer list. Never rendered on DormantTaskRow
 * (Phase 14 compat — dormant surface has no shift semantics).
 */
export function ShiftBadge({
  idealDate,
  scheduledDate,
  timezone,
}: {
  idealDate: Date;
  scheduledDate: Date;
  timezone: string;
}) {
  const idealStr = formatInTimeZone(idealDate, timezone, 'MMM d');
  const scheduledStr = formatInTimeZone(scheduledDate, timezone, 'MMM d');
  const tooltip = `Shifted from ${idealStr} to ${scheduledStr} to smooth household load`;
  return (
    <span
      data-shift-badge
      aria-label="Shifted"
      title={tooltip}
      className="ml-1 inline-flex items-center text-xs"
    >
      ⚖️
    </span>
  );
}
