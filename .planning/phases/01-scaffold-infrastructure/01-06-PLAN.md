```yaml
name: CI

on:
  pull_request:
  push:
    branches: [main]

permissions:
  contents: read

concurrency:
  group: ci-${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint-test-build:
    runs-on: ubuntu-latest
    timeout-minutes: 25
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup Node
        uses: actions/setup-node@v5
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci --no-audit --no-fund

      - name: Lint
        run: npm run lint

      - name: Type check
        run: npm run typecheck

      - name: Unit tests
        run: npm run test

      - name: Install Playwright browsers
        run: npx playwright install --with-deps chromium

      - name: Build
        run: npm run build

      - name: E2E tests
        run: npm run test:e2e

      - name: Set up Buildx (for docker build sanity)
        uses: docker/setup-buildx-action@v4

      # NOTE: --load requires single-platform; see release.yml for multi-arch GHCR push
      - name: Docker build (amd64; --load for local-daemon inspection)
        run: docker buildx build --platform linux/amd64 --load -t homekeep:ci -f docker/Dockerfile .

      - name: Image size check (<300MB per INFR-03)
        run: sh scripts/check-image-size.sh homekeep:ci

      - name: Docker build (arm64 cross-build validation; no --load, Phase-1 success criterion #4)
        run: docker buildx build --platform linux/arm64 --output type=image,push=false -f docker/Dockerfile .

      - name: Regression check - no NEXT_PUBLIC_ baked into client bundle (Pitfall #3)
        run: |
          set -e
          LEAK=$(docker run --rm --entrypoint sh homekeep:ci -c 'grep -rIE NEXT_PUBLIC_ /app/.next 2>/dev/null || true')
          if [ -n "$LEAK" ]; then
            echo "FAIL: NEXT_PUBLIC_* leaked into built bundle:"
            echo "$LEAK"
            exit 1
          fi
          echo "OK: no NEXT_PUBLIC_* in /app/.next"

      - name: Regression check - .env is gitignored
        run: |
          # If someone accidentally committed .env, this catches it
          if git ls-files | grep -E '^\.env$'; then
            echo "FAIL: .env is tracked by git"
            exit 1
          fi
          echo "OK: .env is not tracked"
```

```yaml
name: Release

on:
  push:
    tags: ['v*']

permissions:
  contents: read
  packages: write

concurrency:
  group: release-${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: false

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    timeout-minutes: 60
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Set up QEMU
        uses: docker/setup-qemu-action@v4

      - name: Set up Buildx
        uses: docker/setup-buildx-action@v4

      - name: Log in to GHCR
        uses: docker/login-action@v4
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ghcr.io/${{ github.repository }}
          tags: |
            type=semver,pattern={{version}}
            type=semver,pattern={{major}}.{{minor}}
            type=raw,value=latest,enable={{is_default_branch}}

      - name: Build and push multi-arch
        uses: docker/build-push-action@v7
        with:
          context: .
          file: docker/Dockerfile
          platforms: linux/amd64,linux/arm64
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
          build-args: |
            PB_VERSION=0.37.1
            S6_OVERLAY_VERSION=3.2.2.0

      - name: Verify multi-arch manifest
        run: sh scripts/check-multiarch.sh ghcr.io/${{ github.repository }}:${{ github.ref_name }}
```

| T-01-06-01 | Elevation of Privilege | Malicious PR gains write access to GHCR via ci.yml | mitigate | ci.yml has `permissions: contents: read` only -- no packages:write. Release.yml is the only packages:write workflow and triggers on tags (controlled by maintainers) |
| T-01-06-02 | Tampering | Pinned actions hijacked via upstream tag move | accept | RESEARCH.md pins to major versions (@v4, @v5, @v7) per widespread community convention. SHA pinning is a phase-7 hardening target |
| T-01-06-03 | Information Disclosure | NEXT_PUBLIC_ leaked to built bundle | mitigate | ci.yml step greps built `/app/.next` inside the image for `NEXT_PUBLIC_` strings; any hit fails the workflow (Pitfall #3 phase-1 action) |
| T-01-06-04 | Information Disclosure | `.env` accidentally committed | mitigate | ci.yml step fails if `.env` appears in `git ls-files` |
| T-01-06-05 | Denial of Service | QEMU arm64 build hangs past CI minutes limit | accept | timeout-minutes: 60 caps the damage; Pitfall #6 documents fallback to amd64-only if QEMU is chronically broken; phase 1 stack has no native node modules so QEMU should work |
