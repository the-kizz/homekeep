    ```typescript
    import type { Override } from '@/lib/schedule-overrides';
    import { isOoftTask, computeNextDue, type Task } from '@/lib/task-scheduling';

    export type RebalanceBuckets = {
      anchored: Task[];
      active_snooze: Task[];
      from_now_on: Task[];
      rebalanceable: Task[];
    };

    /**
     * Phase 17 REBAL classifier (D-01 priority order, D-02 exclusions).
     *
     * Priority (first match wins, D-01):
     *   1. anchored       — task.schedule_mode === 'anchored'      (REBAL-01)
     *   2. active_snooze  — overridesByTask.has(task.id)           (REBAL-02)
     *   3. from_now_on    — task.reschedule_marker truthy          (REBAL-03)
     *   4. rebalanceable  — everything else (cycle, no override,
     *                       no marker, non-OOFT, not dormant)      (REBAL-04)
     *
     * Excluded (D-02) — NOT in any bucket:
     *   - task.archived === true
     *   - isOoftTask(task)                     (LOAD-09 — OOFT never smoother-placed)
     *   - dormant-seasonal: computeNextDue returns null for a
     *     natural-only synthesized view of the task (no override,
     *     no smoothed, no marker). If computeNextDue returns null
     *     the task is either archived (already filtered) or
     *     seasonally dormant → exclude from rebalanceable.
     *     Dormant detection MUST run ONLY on the rebalanceable
     *     candidate set — anchored/override/marker tasks are
     *     preserved regardless of dormancy (user intent wins).
     *
     * Pure: no I/O, no Date.now reads internally. Caller supplies
     * `now` and the overridesByTask Map. Completion Map is NOT
     * required by this function — dormant detection synthesizes
     * `lastCompletion = null` (which routes to the seasonal-wakeup
     * branch rather than dormant-null for a truly fresh task; a
     * prior-completion test case below proves the dormant path).
     */
    export function classifyTasksForRebalance(
      tasks: Task[],
      overridesByTask: Map<string, Override>,
      latestByTask: Map<string, { completed_at: string }>,
      now: Date,
      timezone: string = 'UTC',
    ): RebalanceBuckets {
      const buckets: RebalanceBuckets = {
        anchored: [], active_snooze: [], from_now_on: [], rebalanceable: [],
      };
      for (const task of tasks) {
        if (task.archived) continue;
        if (isOoftTask(task)) continue;

        if (task.schedule_mode === 'anchored') { buckets.anchored.push(task); continue; }
        if (overridesByTask.has(task.id))      { buckets.active_snooze.push(task); continue; }
        if (task.reschedule_marker)            { buckets.from_now_on.push(task); continue; }

        // Rebalanceable candidate — check dormant-seasonal exclusion.
        // Synthesize a natural-only task view (no override, no smoothed,
        // no marker) so computeNextDue's seasonal branches report
        // dormancy objectively. If null → dormant, exclude.
        const naturalView: Task = {
          ...task,
          next_due_smoothed: null,
          reschedule_marker: null,
        };
        const last = latestByTask.get(task.id) ?? null;
        const lastCompletion = last ? { completed_at: last.completed_at } : null;
        const due = computeNextDue(naturalView, lastCompletion, now, undefined, timezone);
        if (due === null) continue; // dormant-seasonal (D-02)

        buckets.rebalanceable.push(task);
      }
      return buckets;
    }
    ```
